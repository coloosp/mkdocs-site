    # include<iostream>
# include<string>
using namespace std;
 int main(){
    string s;
    cin >>s;
    int left=s.size()-1;//е·ҰжҢҮй’ҲжҳҜеҺҹжқҘзҡ„ж•°з»„иҫ№з•Ң
    int count =0;
    for(int i=0; i<s.size();i++){//жЈҖжҹҘеҺҹж•°з»„зҡ„ж•°еӯ—дёӘж•°
        if(s[i]>='0' && s[i] <='9')
        count ++;
    }

    s.resize(s.size()+5*count);//жү©е®№sпјҢеӣ дёәжҳҜеҲ еҺ»ж•°еӯ—еңЁеўһеҠ вҖңnumberвҖқ
    int right=s.size()-1;//жү©е®№еҗҺзҡ„ж•°з»„иҫ№з•Ң
    while(left >=0){
        if(s[left]>='0' && s[left] <='9'){//жЈҖжөӢеҲ°жҳҜж•°еӯ—пјҢеҸҚеҗ‘ж·»еҠ "number"
            s[right--]='r';
            s[right--]='e';
            s[right--]='b';
            s[right--]='m';
            s[right--]='u';
            s[right--]='n';
        }
        else{
            s[right--]=s[left];
        }

        left--;
    }
    cout << s<<endl;


 }


 }

    еҜ№дәҺејҖиҫҹж–°ж•°з»„жқҘиҜҙпјҢеҸӘйңҖиҰҒжҲ‘д»¬жЈҖжҹҘеҲ°ж•°еӯ—пјҢе°ұеҗ‘ж–°ж•°з»„йҮҢйқўж·»еҠ "number"пјҢеҸҚд№ӢеҲҷзӣҙжҺҘж·»еҠ еҺҹдҪҚзҪ®е…ғзҙ гҖӮ иҖҢеҜ№дәҺеҸҢжҢҮй’Ҳжі•жқҘиҜҙпјҢжҲ‘д»¬йңҖиҰҒе…ҲжЈҖжөӢеҺҹ

    ж•°з»„пјҢеӣ дёәжҲ‘д»¬жҳҜеңЁеҺҹеӯ—з¬ҰдёІйҮҢеҒҡдҝ®ж”№пјҢйҰ–е…ҲжҳҜзЎ®е®ҡеӯ—з¬ҰдёІзҡ„ж–°й•ҝеәҰпјҢз„¶еҗҺжҲ‘д»¬йңҖиҰҒд»ҺеҗҺеҫҖеүҚжЈҖжөӢпјҢеҪ“жҳҜеӯ—жҜҚе°ұзӣҙжҺҘж·»еҠ пјҢжЈҖжөӢеҲ°ж•°еӯ—еҲҷйңҖиҰҒеҸҚеҗ‘ж·»еҠ "number"гҖӮ дёәд»Җд№ҲдёҚжҳҜд»ҺеүҚеҫҖеҗҺе‘ўпјҹеӨ§е®¶еҸҜд»ҘиҜ•жғіпјҢд»ҺеүҚеҫҖеҗҺпјҢжҲ‘д»¬жҜҸж¬ЎжЈҖжөӢеҲ°ж•°еӯ—пјҢе°ұйңҖиҰҒжҠҠеҗҺйқўзҡ„еӯ—жҜҚз§»еҠЁпјҢиҝҷж ·дјҡжһҒеӨ§еўһеҠ ж—¶й—ҙеӨҚжқӮеәҰ
