# 统计坏数对的数目
  这里是非黑即白的世界，所以除了好就是坏！思路很明显，我们统计好数组的数目，用数组对数总和减去好数对即可
  统计好数对的方式类似于两数之和
```
class Solution {
public:
    long long countBadPairs(vector<int>& nums) {
        unordered_map <long long,long long> map;
        long long goods=0;
        for(int i=0;i<nums.size();i++){
            goods+=map[i-nums[i]];
            map[i-nums[i]]++;
        }

        
        long long result=(nums.size()* (nums.size()-1))/2 -goods;
        return result;
    }
};
```
