```
#include <iostream>
# include <vector>
# include <queue>
#include <algorithm>
# include <unordered_map>
# include <unordered_set>
// #include <bits/stdc++.h>
using namespace std;


int main() {
	int n, k, q;
	cin >> n >> k >> q;
	int hash[200005] = {0};
	while (n--) {
		int l, r;
		cin >> l >> r;
		for (int i = l; i <= r; i++) {
			hash[i]++;
		}
	}

	while (q--) {
		int count = 0;
		int a, b;
		cin >> a >> b;
		for (int i = a; i <= b; i++) {
			if (hash[i] >= k) {
				count++;
			}
		}
		cout << count << endl;
	}


	return 0;
}
```
