# 幸运草（贪心）
[幸运草](https://www.luogu.com.cn/problem/P11642)
哥们取的名字和内容有啥关系吗？
好吧，和前面的题目一样，要贪才能赢！
```
#include <iostream>
# include <vector>
# include <queue>
// #include <bits/stdc++.h>
using namespace std;

int main() {
	long long n, a;
	cin >> n >> a;
	long long pre[100005];
	long long ans = 0;
	for (long long i = 0; i < n; i++) {
		int b;
		cin >> b;
		ans += b;
		pre[i] = a - b;
	}

	long long result = -1;
	long long count = 0;
	for (long long i = 0; i < n; i++) {
		count += pre[i];
		if (count < 0) {
			count = 0;
		}
		if (count > result) {
			result = count;
		}
	}

	cout << ans + result << endl;







	return 0;
}
```
