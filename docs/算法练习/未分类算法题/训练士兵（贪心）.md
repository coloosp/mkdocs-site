#include <iostream>
# include <vector>
# include <queue>
#include <algorithm>

using namespace std;
bool compare(const pair<long long,long long>& a, const pair <long long,long long>& b) {
	return a.second < b.second;
}


int main() {
	long long n, s;
	cin >> n >> s;
	vector <pair<long long, long long>> soldier(n);
	long long ans = 0;
	for (long long i = 0; i < n; i++) {
		cin >> soldier[i].first >> soldier[i].second;
		ans += soldier[i].first;
	}

	sort(soldier.begin(), soldier.end(), compare);
	
	long long result = 0;
	long long tk = 0;
	for (long long i = 0; i < n; i++) {
		if (ans >= s) {
			result += (soldier[i].second - tk) * s;
			ans -= soldier[i].first;
			tk += soldier[i].second -tk ;
		}
		else {
			result += (soldier[i].second - tk) * soldier[i].first;
			ans -= soldier[i].first;
		}



	}


	cout << result <<endl;

	return 0;
}
