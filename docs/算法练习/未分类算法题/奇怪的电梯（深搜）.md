# 奇怪的电梯（深搜）
[奇怪的电梯](https://www.luogu.com.cn/problem/P1135)
优化的深搜，在原始版本里面我们把图存储在一个数组里面，实际上我们可以存储两个数组，1：每一层楼到达的最小步数，2：每一层楼允许到达的楼层

在深搜里面，我们进行了优化，只有当前步数+1小于所要到达楼层的步数我们才进入下一步深搜，嗯，只有优化...
```
#include <iostream>
# include <vector>
# include <queue>
using namespace std;
int n;

void dfs(vector<int> &floor,vector <int> &dis,int s,int t) {
	floor[s] = t;

	if (s + dis[s] <= n && t + 1 < floor[s + dis[s]]) {
		dfs(floor, dis, s + dis[s], t + 1);
	}

	if (s - dis[s] > 0 && t + 1 < floor[s - dis[s]]) {
		dfs(floor, dis, s - dis[s], t + 1);
	}

	return;
}




int main() {
	int  a, b;
	cin >> n >> a >> b;
	vector <int> floor(205, INT_MAX);
	vector <int> dis(205);


	for (int i = 1; i<=n; i++) {
		cin >> dis[i];
	}
	
	dfs(floor, dis, a, 0);

	if (floor[b] != INT_MAX) {
		cout << floor[b] << endl;
	}
	else {
		cout << -1 << endl;
	}

	
	








	return 0;
}
```
