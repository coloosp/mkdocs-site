# 挖地雷（多起点深搜）
[挖地雷](https://www.luogu.com.cn/problem/P2196)
好吧，就是深搜，不过确实没有反应过来。因为题目给出的是可以从多个起点开始搜索，我们知道深搜虽然会退回，但是最多最多退回到起点，所以我们需要在调用函数时就多起点开始深搜

```
#include <iostream>
# include <vector>
# include <queue>
#include <algorithm>
# include <unordered_map>
# include <unordered_set>
// #include <bits/stdc++.h>
using namespace std;
int n;
vector <int> path;
vector <int> result_path;
vector<vector<int>> graph(21, vector<int>(21, 0));
vector <int> b(21, 0);

int result=0;
int sum=0;

int book[21];

bool check(int x) {
	bool flag = false;
	for (int i = x+1; i <= n; i++) {
		if (graph[x][i] == 1 ) {
			flag = true;
		}

	}
	return flag;
}

void dfs(int index) {

	if (check(index)== false) {
		if (sum > result) {
			result_path = path;
			result = sum;
		}
	}


	for (int i = 1; i <= n; i++) {
		if (book[i]) {
			continue;
		}

		if (!graph[index][i]) {
			continue;
		}

		book[i] = 1;
		path.push_back(i);
		sum += b[i];

		dfs(i);

		path.pop_back();
		sum -= b[i];
		book[i] = 0;
	}


}



int main() {
	cin >> n;

	for (int i = 1; i <=n; i++) {
		cin >> b[i];
	}
	for (int i = 1; i <n; i++) {
		for (int j = 1; j <=n-i; j++) {
			int a;
			cin >> a;
			graph[i][i + j]=a;
		}
	}

	for (int i = 1; i <=n; i++) {
		graph[i][i] = 1;
	}

	for (int i = 1; i <= n; i++) {
		dfs(i);
	}//关键，我们需要多起点搜索

	for (int i = 0; i < result_path.size(); i++) {
		cout << result_path[i] <<" ";
	}
	cout << endl;
	cout << result;


}
```
