# 填涂颜色（广搜）
[填涂颜色](https://www.luogu.com.cn/problem/P1162)
思路是将边界先暂时改成其他数字，比如3，然后再在里面的0改成2，然后复原即可

```
// #include <bits/stdc++.h>
using namespace std;
int dir[4][2] = { 0,1,1,0,0,-1,-1,0 };
void bfs(vector<vector <int>> &graph,int x,int y) {
	queue <pair<int, int>> que;
	que.push({ x,y });
	graph[x][y] = 3;
	while (!que.empty()) {
		int x = que.front().first;
		int y = que.front().second;
		que.pop();
		for (int i = 0; i < 4; i++) {
			int next_x = x + dir[i][0];
			int next_y = y + dir[i][1];
			if (next_x < 0 || next_x >= graph.size() || next_y < 0 || next_y >= graph.size()) {
				continue;
			}

			if (graph[next_x][next_y] == 0) {
				que.push({ next_x,next_y });
				graph[next_x][next_y] = 3;
			}
		}
	}
}



int main() {
	int n;
	cin >> n;

	vector <vector <int>> graph(n, vector <int>(n, 0));
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			cin >> graph[i][j];
		}
	}

	for (int i = 0; i < n; i++) {
		if (graph[i][0] == 0) bfs(graph, i, 0);
		if (graph[i][n - 1] == 0) bfs(graph, i, n - 1);
		if (graph[0][i] == 0) bfs(graph, 0, i);
		if (graph[n - 1][i] == 0) bfs(graph, n - 1, i);
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			if (graph[i][j] == 0) {
				graph[i][j] = 2;
			}
			if (graph[i][j] == 3) {
				graph[i][j] = 0;
			}
		}
	}



	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			cout << graph[i][j] << " ";
		}
		cout << endl;
	}








	return 0;
}
```
