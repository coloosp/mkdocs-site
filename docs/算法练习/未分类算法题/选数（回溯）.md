# 选数（回溯）

深搜练手题
```
#include <iostream>
# include <vector>
# include <queue>
// #include <bits/stdc++.h>
using namespace std;
int n;
int path;
vector<long long> result;
int sum = 0;
bool isprime(long long n) {
	if (n <= 1) {
		return false;
	}
	for (long long i = 2; i * i < n; i++) {
		if (n % i == 0) {
			return false;
		}


	}
	return true;

}

void backtracking(vector<long long>& nums, int index, int k) {
	if (path == k) {
		result.push_back(sum);
	}

	for (int i = index; i < n; i++) {
		path++;
		sum += nums[i];
		backtracking(nums, i + 1, k);
		path--;
		sum -= nums[i];
	}
}





int main() {
	int k;
	cin >> n >> k;
	vector <long long> nums(n);
	for (int i = 0; i < n; i++) {
		cin >> nums[i];
	}

	int ans = 0;
	backtracking(nums, 0, k);


	for (int i = 0; i < result.size(); i++) {
		if (isprime(result[i])) {
			ans++;
		}
	}

	cout << ans << endl;




	return 0;
}
```
