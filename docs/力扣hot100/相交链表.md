```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        int lenA=0;
        int lenB=0;
        ListNode *p=headA->next;
        while(p!=NULL){ //统计A链表长度
            lenA++;
            p=p->next;
        }

        p=headB->next;
        while(p!=NULL){ //统计B链表长度
            lenB++;
            p=p->next;
        }

        ListNode*pa=headA;
        ListNode*pb=headB;
        //让长的链表先走差值距离
        if(lenA >lenB){
            int dist=lenA-lenB;
            while(dist--)
                pa=pa->next;
        }
        else{
            int dist=lenB-lenA;
            while(dist--)
                pb=pb->next;
        }

        //检查是否有相同节点
        while(pa!=NULL && pb!=NULL){
            if(pa==pb)
                return pa;
            pa=pa->next;
            pb=pb->next;
        }

        return NULL;

    }
};
```

时间复杂度O(n)
空间复杂度O(1)
