# 翻转二叉树
[翻转二叉树](https://leetcode.cn/problems/invert-binary-tree/?envType=study-plan-v2&envId=top-100-liked)

其实思路很简单，各位观察例子就知道，如果把每个节点的左右节点都交换一遍就可以完成翻转二叉树了

根据这样的思路我们来完成前序遍历形式的翻转二叉树

## 递归
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void swapTree(TreeNode*&node)
    {
        if(node==nullptr)
            return ;
        swap(node->left,node->right);//交换左右子树
        swapTree(node->left);//翻转左子树
        swapTree(node->right);//翻转右子树
    }
    TreeNode* invertTree(TreeNode* root) {
         swapTree(root);
         return root;
    }
};
```

实际上写成后序遍历也完全没有问题，因为我们只需要让所有节点的左右节点交换即可

```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void swapTree(TreeNode*&node)
    {
        if(node==nullptr)
            return ;
        swapTree(node->left);//翻转左子树
        swapTree(node->right);//翻转右子树
        swap(node->left,node->right);//交换左右子树，只是把这个放到了后面
    }
    TreeNode* invertTree(TreeNode* root) {
         swapTree(root);
         return root;
    }
};
```

时间复杂度O(n)
空间复杂度O(logn)

## 层序遍历
好吧本来是后面的内容，不过可以使用我就顺便写了
因为层序遍历属于经典算法，也就不啰嗦解释了
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        queue <TreeNode*> Q;
        if(root==nullptr)
            return root;
        TreeNode* t;//暂存指针
        Q.push(root);
        while(!Q.empty())
        {
            t=Q.front();
            Q.pop(); //获取并弹出队首
            if(t->left) Q.push(t->left);
            if(t->right) Q.push(t->right);
            swap(t->left,t->right); //这里放在if前面也没有问题
        } 
        return root;
    }
};
```
