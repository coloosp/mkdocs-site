# 矩阵置零
[矩阵置零](https://leetcode.cn/problems/set-matrix-zeroes/description/?envType=study-plan-v2&envId=top-100-liked)

## 朴素模拟
好吧这样做实际上和单独做一个二维数组没什么区别，而且很简单好想，稍微看看就行，毕竟这是中等题，不太可能这么简单

```
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        //这里因为要求的原因，我们实际上不能使用任何数字来标记
        //我的思路是使用一个数组/栈来存储零的位置，但是这样其实是有问题的，因为它可能是最坏的O(mn)空间复杂度
        vector <vector<int>> mark;


        for(int i=0;i<matrix.size();i++){
            for(int j=0;j<matrix[0].size();j++){
                if(matrix[i][j]==0)
                    mark.push_back({i,j});
            }
        }

        for(vector <int> tmp : mark){
            int row=tmp[0];
            int col=tmp[1];
            for(int i=0;i<matrix[0].size();i++)
                matrix[row][i]=0;
            for(int i=0;i<matrix.size();i++)
                matrix[i][col]=0;
        
        }




        
    }
};
```
时间复杂度O(mn)
空间复杂度O(mn)最坏情况是全部都是0

## 脑筋急转弯
答案是使用数组本身来作为标记数组，有趣吧，就像上一题一样，原地哈希表，就是把数组本身作为了哈希表
```
class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        //这实际上更像是脑筋急转弯，我们会这样做
        //1、我们检查第一行与第一列的0元素存入数组中
        //2、检查除了第一行与第一列以外的其他0元素，并且在对应第一行与第一列标记
        //3、除了第一行与第一列的0元素将该列/行全部标零
        //4、然后我们再处理数组中的存入的本来就是第一行第一列的元素即可

        vector <vector<int>> mark;//标记本来就是第一行或者第一列的元素
        for(int i=0;i<matrix.size();i++){//检查i行第一列
            if(matrix[i][0]==0)
                mark.push_back({i,0});
        }

        for(int i=0;i<matrix[0].size();i++){//检查i列第一行
            if(matrix[0][i]==0)
                mark.push_back({0,i});
        }

        for(int i=1;i<matrix.size();i++){ //检查内部的0元素
            for(int j=1;j<matrix[0].size();j++){
                if(matrix[i][j]==0) //将其对应行与列标记
                {
                    matrix[i][0]=0;
                    matrix[0][j]=0;
                }
            }
        }

        for(int i=1;i<matrix.size();i++){ //这里一定要注意不能检查第一行
            if(matrix[i][0]==0)
            {
                for(int j=1;j<matrix[0].size();j++)
                    matrix[i][j]=0;
            }
        }

        for(int j=1;j<matrix[0].size();j++){ //同样不能检查第一列
            if(matrix[0][j]==0)
            {
                for(int i=1;i<matrix.size();i++)
                    matrix[i][j]=0;
            }
        }

            for(vector <int> x : mark){ //除了本来存在于第一行与第一列的0
            int row=x[0];
            int col=x[1];
            for(int i=0;i<matrix[0].size();i++)
                matrix[row][i]=0;
            for(int i=0;i<matrix.size();i++)
                matrix[i][col]=0;
        }


        

    }
};
```
