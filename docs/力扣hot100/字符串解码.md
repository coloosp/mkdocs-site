这就是为什么作者很讨厌字符串的题了
你觉得你可以做得出来，但是你真的可以做得出来吗？
这种题就相当坏了

好吧，这道题的巧思还不少，利用栈的思想来处理，并且有很多坑人的地方

## 解析
比如我们的数字不一定是个位，所以如何保留一个完整的数字是个问题

比如有点字符串没有用[]包裹，也不需要数字

比如拥有嵌套结构内层的解码完了会变成外层的字符串值

我的思路就是利用双栈结构
一个为数字栈st_num，一个是字符栈st_ch

接下来我们会遇到以下几种情况，请小心区分
1. 遇到数字，请不要直接加入到数字栈st_num，我们应该让num=num*10+(c-'0')，直到符合条件时再加入
2. 遇到字符时，直接加入字符栈st_ch即可，无论前面是否有[
3. 遇到[左括号时，说明前面的数字已经收集完了，加入数字栈st_num中即可，左括号加入字符栈st_ch中
4. 遇到右括号，要开始正经干活了，这种情况一定是要重复k次的加入字符串，首先获取需要重复的字符串直到在字符栈st_ch中遇到'['，并且将它们都弹出。取出数字栈st_num中的数值k，重复k次加入到字符栈st_ch中。要小心我们栈是倒着加入的，需要反转回来

将字符栈st_ch中的所有字符加入到结果字符串中，注意这里也是倒着的，要反转过来

## 代码

```
class Solution {
public:
    string decodeString(string s) {
        // 字符串题素质不是很高啊
        // 要实现这道题需要使用双栈结构，一个栈存储数字，一个栈存储字符
        string result="";
        stack<int> st_num;
        stack<char> st_ch;
        int num = 0; // 存储数值用
        for (char c : s) {
            if (c >= '0' && c <= '9') {
                // 一定不要直接加入栈，我们不知道这个值是否是个位数
                num = num * 10 + (c - '0');
            } else if (c >= 'a' && c <= 'z')
                st_ch.push(c);
            else if (c == '[') {
                st_ch.push(c);
                // 到这里数字部分已经结束了
                st_num.push(num);
                num = 0; // 记得重置！
            } else       // 检查到']'了
            {
                string temp = "";
                char t = st_ch.top();
                while (t != '[') {
                    st_ch.pop();
                    temp += t;
                    t = st_ch.top();
                }

                st_ch.pop(); // 弹出最后一个'['

                reverse(temp.begin(), temp.end());

                //获取需要加入的次数
                int k = st_num.top(); // 这里数字栈st_num一定有值的
                st_num.pop();

                // 将temp字符串重复k次加入字符栈st_ch
                for (int i = 0; i < k; i++) {
                    for (int j = 0; j < temp.size(); j++) {
                        st_ch.push(temp[j]);
                    }
                }
            }
        }

        while (!st_ch.empty()) {
            char ch = st_ch.top();
            result += ch;
            st_ch.pop();
        }

        reverse(result.begin(), result.end());
        return result;
    }
};
```

时间复杂度O(n)
空间复杂度O(n)

当然我们C++的字符串本身就可以当做栈，可以进行一些小小的优化

public:
    string decodeString(string s) {
        // 字符串题素质不是很高啊
        // 要实现这道题需要使用双栈结构，一个栈存储数字，一个栈存储字符
        string result="";//将结果字符串本身就作为栈使用
        stack<int> st_num;
        int num = 0; // 存储数值用
        for (char c : s) {
            if (c >= '0' && c <= '9') {
                // 一定不要直接加入栈，我们不知道这个值是否是个位数
                num = num * 10 + (c - '0');
            } else if (c >= 'a' && c <= 'z')
                result+=c;
            else if (c == '[') {
                result+=c;
                // 到这里数字部分已经结束了
                st_num.push(num);
                num = 0; // 记得重置！
            } else       // 检查到']'了
            {
                string temp = "";
                char t = result.back();
                while (t != '[') {
                    result.pop_back();
                    temp += t;
                    t = result.back();
                }

                result.pop_back(); // 弹出最后一个'['

                reverse(temp.begin(), temp.end());

                //获取需要加入的次数
                int k = st_num.top(); // 这里数字栈一定有值的
                st_num.pop();

                // 将temp字符串重复k次加入字符栈
                for (int i = 0; i < k; i++) {
                   result+=temp;
                }
            }
        }

        return result;
    }
};
