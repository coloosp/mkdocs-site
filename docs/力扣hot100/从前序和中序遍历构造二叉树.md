# 从前序和中序遍历构造二叉树
[从前序和中序遍历构造二叉树](https://leetcode.cn/problems/construct-binary-tree-from-preorder-and-inorder-traversal/?envType=study-plan-v2&envId=top-100-liked)

令人怀念的题目呢，这是作者学校考研的自命题倒数第二题，思路和当时一样，不知道细节方面处理正确没有

各位如果学习过数据结构（怎么又在说数据结构？）,一定做过无数遍这种小题，也就是利用前序/后序/层序 +中序手搓二叉树

事实也如此，结论是**任意序列+中序可以确定一个二叉树**

不过教材里面很少提及原因，大家都是凭感觉（至少我是）画出二叉树

我给出原因，因为中序遍历是左中右，利用中序遍历可以确定节点是处于左子树还是右子树，所以这里的思路就是使用前序序列构造二叉树并使用中序遍历确定下一个节点是否插入左子树或右子树

我们的前序序列是中左右，在插入节点后下一个节点会尽可能选择左边**直到左边不能选择**，什么叫左边不能选择？也就是在中序遍历里面已经不存在这个值了！

深呼吸，我们凑齐了所有碎片，来试着构造出答案吧！
1. 实现一个搜索函数，用来确定对应值在中序中处于的位置index
2. 设置计数count，用来确定我们当前节点应该选择前序序列中的哪一个
3. 实现构造二叉树函数，这是一个递归函数，遵循递归三步  
* 因为要构造树所以传入节点引用node，传入low与high确定当前子树是否还有节点，传入前序和中序，传入count计数
* 当low >high是说明当前子树已经没有节点需要构造了，将传入的node设置为空并返回
* 构造当前节点并根据count传入前序的值，确定值val在中序中的位置index，利用index来构造其左右子树

## 原始代码
不算优美但是完成了
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
public:
    int search(int val, vector<int>& inorder) { //搜索对应值在inorder的位置
        for (int i = 0; i < inorder.size(); i++) {
            if (inorder[i] == val)
                return i;
        } 
        return 0;
    }
    void plantTree(TreeNode*& node, int low, int high, vector<int>&preorder,
                   vector<int>&inorder, int&count) {
        if (low > high) { //说明已经不存在需要构造的节点了
            node = nullptr; // 置空
            return;
        }
        node = new TreeNode(preorder[count++]);//注意要在这里自增而不是函数内
        int index = search(node->val, inorder);
        plantTree(node->left, low, index-1, preorder, inorder, count);
        plantTree(node->right, index+1, high, preorder, inorder, count);
    }
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        // 老朋友呢
        // 思路是按照先序构造，按照中序判断左右
        int high = preorder.size()-1;
        if (high == -1)
            return nullptr;
        int low = 0;
        TreeNode* root = nullptr;
        int count=0;
        plantTree(root, low, high, preorder, inorder, count);
        return root;
    }
};
```

时间复杂度O(n²)
空间复杂度O(1) 不算上必要开销

## 哈希优化
利用哈希表<值，下标>的方式来规避每一次都进行的位置确定
```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
public:
    void plantTree(TreeNode*& node, int low, int high, vector<int>&preorder,
                   unordered_map<int,int>&hashmap, int&count) {
        if (low > high) { //说明已经不存在需要构造的节点了
            node = nullptr; // 置空
            return;
        }
        node = new TreeNode(preorder[count++]);//注意要在这里自增而不是函数内
        int index =hashmap[node->val];
        plantTree(node->left, low, index-1, preorder, hashmap, count);
        plantTree(node->right, index+1, high, preorder, hashmap, count);
    }
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        //试着使用哈希表优化吧!
        int high = preorder.size()-1;
        if (high == -1) //特判
            return nullptr;
        int low = 0;
        unordered_map <int,int> hashmap;//为<值，下标>的哈希表
        for(int i=0;i<inorder.size();i++)
        {
            hashmap[inorder[i]]=i;
        }
        TreeNode* root = nullptr;
        int count=0;
        plantTree(root, low, high, preorder, hashmap, count);
        return root;
    }
};
```
时间复杂度O(n)
空间复杂度O(n)
