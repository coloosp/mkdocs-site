```
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        //实现起来应该不难，如果找到则返回mid，如果找不到则返回low
        int low=0;
        int high=nums.size()-1;
        while(low <=high)
        {
            int mid=(low+high)/2;
            if(nums[mid]==target)
                return mid;
            if(nums[mid] >target)
                high=mid-1;
            else
                low=mid+1;
        }

        return low;
    }
};
```

时间复杂度O(logn)
空间复杂度O(1)
