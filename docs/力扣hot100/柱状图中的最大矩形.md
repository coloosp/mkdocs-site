# 柱状图中的最大矩形
[柱状图中的最大矩形](https://leetcode.cn/problems/largest-rectangle-in-histogram/?envType=study-plan-v2&envId=top-100-liked)

## 暴力算法
我们先尝试一下暴力算法吧，思路就是记录每一个节点的高度，然后向前向后探索，直到遇到比该柱子更小的高度为止，简单好想的思路，可惜过不了

也不要灰心，暴力算法会给我们后续的算法提示，并非毫无意义

```
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        //让我们先尝试暴力算法吧
        vector <int> result(heights.size(),0);
        for(int i=0;i<heights.size();i++)
            {
                result[i]+=heights[i];
                //然后向前向后探索，直到遇到比该柱子高度小的
                for(int j=i+1;j<heights.size();j++)
                    {
                        if(heights[j] <heights[i])
                            break;
                        
                        result[i]+=heights[i];
                    }
                for(int j=i-1;j>=0;j--)
                    {
                        if(heights[j] <heights[i])
                            break;
                        
                        result[i]+=heights[i];
                    }
            }
            int maxarea=0;
            //搜索最大面积
            for(int i=0;i<result.size();i++)
                {
                    maxarea=max(result[i],maxarea);
                }

            return maxarea;
    }
};
```
时间复杂度O($n^2$)
空间复杂度O(n)
## 单调栈
各位能从上面的暴力算法中思考到什么呢？我们核心需要检查的仅仅是左边和右边比该柱子矮，比如我们从左往右检查时，我们如果遇到比当前柱子矮的柱子，我们要做的事情就是通知当前柱子收集结果后洗洗睡了(弹出栈)
叫单调栈也就是因为我们会维护栈内从栈底到栈顶的单调递增
至于为什么我们需要存储下标相信各位在前面每日温度应该知道原因1

```
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        //接下来尝试一下单调栈的做法吧
        //搜索面积的本质就是找到左边和右边最近的比该点低的柱子
        int n=heights.size();
        stack <int> st;//单调栈，注意存储的是下标
        vector <int> result(heights.size(),0);

        //先从左往右搜索
        st.push(0);
        for(int i=1;i<heights.size();i++)
        {
            while(!st.empty() && heights[st.top()] > heights[i])
                {
                    result[st.top()]+=heights[st.top()]*(i-st.top());
                    st.pop();
                }
            st.push(i);
        }

        //最后往栈里面加入0
        while(!st.empty())
        {
            result[st.top()]+=heights[st.top()]*(n-st.top());
            st.pop();
        }

        //然后从右往左搜索
        st.push(n-1);
        for(int i=n-2;i>=0;i--)
        {
            while(!st.empty() && heights[st.top()] > heights[i])
            {
                result[st.top()]+=heights[st.top()] *(st.top()-i);
                st.pop();
            }
            st.push(i);
        }

        while(!st.empty())
        {
            result[st.top()]+=heights[st.top()] *(st.top()+1);
            st.pop();
        }

        int maxarea=0;

        for(int i=0;i<result.size();i++)
        {
            //减去重复计算的一份
            result[i]-=heights[i];
            maxarea=max(result[i],maxarea);
        }

        return maxarea;
    }
};
```

时间复杂度O(n)
空间复杂度O(n)

上面是按照暴力算法思路得出的单调栈算法，我们还可以对其优化

各位再思考一下，单调栈在干什么？答案是维护单调栈的单调有序性(单调栈一定要单调)，在从左往右的搜索中如何找到当前柱子最近左端比它矮的柱子呢？答案是救赎之道就在其中！我们的单调栈本身就是单调递增的，也就是说下一个栈顶元素就是最近左端,也就是说我们的宽度可以这样计算
`当前搜索到符合条件的i-栈顶弹出后的栈顶值-1`

```
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        // 试着优化单调栈吧
        stack<int> st;
        heights.insert(heights.begin(), 0); // 方便计算宽度
        heights.push_back(0);               // 方便我们置0处理

        st.push(0);
        st.push(1);
        int result = 0;
        for (int i = 2; i < heights.size(); i++) {
            while (!st.empty() && heights[i] < heights[st.top()]) {
                int mid = st.top();
                st.pop();
                int left = st.top(); // 这是因为单调栈的特性，左端就是
                int right = i;
                int wid = right - left - 1; // 计算时是左开右开区间
                int area = heights[mid] * wid;
                result = max(area, result); //可以实时更新不需要另外存储了
            }
            st.push(i);
        }

        return result;
    }
};
```
