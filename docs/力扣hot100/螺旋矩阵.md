唯一的问题是一种特判，对于奇数边的正方形，有一个唯一中心点需要处理

```
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        //一道很考验边界处理的题目
        //需要冷静思考设置的初始条件,设置出发点startx与starty,我们每一圈实际上是进行了四个方向，为了避免冲突，每一次走m-offset与n-offset步
        //这里因为是矩阵，我们可以通过计数来判断什么时候停止
        //搜索时可以使用i,j来作为工作指针

        int m=matrix.size();
        int n=matrix[0].size();
        int count=0;
        int startx=0;
        int starty=0;
        int offset=1; //偏移量
        vector <int> result;
        while(count <=m*n){
            int j=startx;
            int i=starty;
            for(;j<n-offset;j++){//→
                result.push_back(matrix[i][j]);
                count++;
                if(count>=m*n)
                    return result;
            }

            for(;i<m-offset;i++){//↓
                result.push_back(matrix[i][j]);
                 count++;
                if(count>=m*n)
                    return result;
            }

            for(;j>startx;j--){//←
                result.push_back(matrix[i][j]);
                count++;
                if(count>=m*n)
                    return result;
            }

            for(;i>starty;i--){//↑
                result.push_back(matrix[i][j]);
                count++;
                if(count>=m*n)
                    return result;
            }

            startx++;
            starty++;
            offset++;

            if(count==m*n-1 && (m==n && m%2==1)){ //这是唯一特判，也就是奇数边的正方形
                result.push_back(matrix[m/2][n/2]);
                break;
            }
                
        }

        return result;
    }
};
```

时间复杂度O(mn)
空间复杂度O(1)

class Solution {
public:
    vector<vector<int>> generateMatrix(int n) {
        vector<vector<int>> matrix(n,vector<int>(n));
        int loop=n/2;//循环次数，因为是正方形变得可计算了
        int startx=0;
        int starty=0;
        int offset=1;
        int count=1;//标记数

        while(loop--){
            int j=startx;//横向工作指针
            int i=starty;//纵向工作指针

            for(;j<n-offset;j++){//→
                matrix[i][j]=count++;
            }

            for(;i<n-offset;i++){//↓
                matrix[i][j]=count++;
            }

            for(;j>startx;j--){//←
                matrix[i][j]=count++;
            }

            for(;i>starty;i--){//↑
                matrix[i][j]=count++;
            }

            startx++;
            starty++;
            offset++;

        }
            if(n%2==1){//为了处理奇数边的正方形最后一格找不到
                matrix[n/2][n/2]=count;
            }

            return matrix;




    }
};
