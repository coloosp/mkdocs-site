# 最小覆盖子串
[最小覆盖子串](https://leetcode.cn/problems/minimum-window-substring/description/?envType=study-plan-v2&envId=top-100-liked)

作者使用了和前面一样的套路实现的，如果案例不是要求大小写都有一个还算简洁，不过这写的是有点难受，我们把大写的映射到了从26下标开始
```
class Solution {
public:
    string minWindow(string s, string t) {
        //或许可以使用滑动窗口+哈希表来完成
        //我们设置left,right,以及两个hash表，这里居然是大写小写都有，太恶魔了
        //不要学习我的代码，一定还有其他方法解决，这只是滑动窗口的死板写法
        if(t.empty())
            return  "";
        if(s.size() <t.size())
            return ""; //剪枝

        int left=0;
        string result="";
        int matchnum=0;
        int minsize=INT_MAX;
        int result_left=left;
        int result_right=left;//只是初始化

        vector <int> thash(52,0);//记录t的哈希表后续不会初始化后不会改动
        vector <int> windowshash(52,0);//双哈希表
        for(int i=0;i<t.size();i++)
        {
           if(t[i]>='a' && t[i] <='z') 
             thash[t[i]-'a']++;//哈希统计
            else{
                thash[t[i]-'A'+26]++; //大写字母放在后面
            }
        }

        for(int right=0;right <s.size();right++){
           if(s[right]>='a' && s[right] <='z'){ //记录右窗口
             windowshash[s[right]-'a']++;
            if(windowshash[s[right]-'a'] <=thash[s[right]-'a'])
                matchnum++;
           }
            else{
                windowshash[s[right]-'A'+26]++;
                if(windowshash[s[right]-'A'+26]  <=thash[s[right]-'A'+26] )
                    matchnum++;
            }

            while(matchnum==t.size()){ //完成匹配时
                if(right-left+1 < minsize){
                    minsize=right-left+1;
                    result_left=left;
                    result_right=right;
                }

                if(s[left]>='a' && s[left] <='z'){
                    if(windowshash[s[left]-'a']<=thash[s[left]-'a'])
                        matchnum--;
                windowshash[s[left]-'a']--;
                left++; //移动左窗口直到不能移动
                }
                else{
                    if(windowshash[s[left]-'A'+26] <=thash[s[left]-'A'+26])
                        matchnum--;
                windowshash[s[left]-'A'+26]--;
                left++;
                }
            }

        }

        if(minsize==INT_MAX)
            return "";
        result = s.substr(result_left, result_right - result_left + 1);

        return result;







    }
};
```

时间复杂度O(m+n)
空间复杂度O(m+n)
