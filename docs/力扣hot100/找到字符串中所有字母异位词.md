# 找到字符串中所有字母异位词
[找到字符串中所有字母异位词](https://leetcode.cn/problems/find-all-anagrams-in-a-string/description/?envType=study-plan-v2&envId=top-100-liked)
使用了滑动窗口+哈希表的形式来完成
```
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        // 1、建立滑动窗口，left=0.right=p.size()-1,标记匹配数matchnum
        // 2、建立两个哈希表，一个计算p中出现的字母数，一个计算s中匹配的对应字母
        // 3、每一次成功匹配时加入left指针，右移left指针，如果之前是匹配字母就让matchnum--,右移right指针，如果存在匹配字母就让matchnum++

        vector<int> result;
        if(s.size()<p.size())//剪枝
            return result;
        int left = 0;
        int right = p.size() - 1;
        vector<int> windowshash(26, 0);
        vector<int> phash(26,0);
        int matchnum = 0;

        for (int i = 0; i < p.size(); i++) {
            phash[p[i] - 'a']++;
        }

        for (int i = 0; i < p.size(); i++) {
            windowshash[s[i]-'a']++;
            if(windowshash[s[i]-'a']  <=phash[s[i]-'a'] )
                matchnum++;
        }

        while (right < s.size()) {
            if (matchnum == p.size())
                result.push_back(left);

            if(windowshash[s[left]-'a'] <=phash[s[left]-'a'])
                matchnum--;

            windowshash[s[left]-'a']--;//弹出左窗口

            left++;
            right++; // 窗口滑动
            if(right==s.size())//这里是写法问题，应该可以避免这种break
                break;
             windowshash[s[right]-'a']++;//加入右窗口
            if (windowshash[s[right] - 'a'] <=phash[s[right]-'a']) 
                matchnum++;
    
        }

        return result;
    }
};
```

时间复杂度O(n)
空间复杂度O(1)


## 优化版本
各位嫌麻烦可以直接看这个版本，作者忘记了vector可以直接比较是否相等
思路完全一致，都是定长滑动窗口
```
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        vector <int> phash(26,0);
        vector <int> shash(26,0);
        vector <int> result;
        if(s.size() <p.size())
            return result;

        for(int i=0;i<p.size();i++)
        {
            phash[p[i]-'a']++;
        }

        int left=0;
        int right=0;
        //先扩充到对应长度,注意这里-1是为了后面循环更好写
        for(;right<p.size()-1;right++)
        {
            shash[s[right]-'a']++;
        }

        for(;right<s.size();right++)
        {
            shash[s[right]-'a']++;
            if(shash==phash) //哇，可以直接比较诶
                result.push_back(left);
            shash[s[left]-'a']--;
            left++;
        }

        return result;

    }
};
```
