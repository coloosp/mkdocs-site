# 二叉树展开成链表
[二叉树展开成链表](https://leetcode.cn/problems/flatten-binary-tree-to-linked-list/description/?envType=study-plan-v2&envId=top-100-liked)

允许额外空间的话应该有各种方法可以实现，各位也可以不看题解自行完成，应该在简单题的难度（链表+前序遍历）

```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void preOrder(TreeNode* node,queue<TreeNode*>&Q)
    {
        if(node==nullptr)
            return;
        Q.push(node);
        preOrder(node->left,Q);
        preOrder(node->right,Q);
    }
    void flatten(TreeNode* root) {
        queue <TreeNode*> Q;
        preOrder(root,Q);
        TreeNode dummyhead;
        TreeNode* r=&dummyhead;
        TreeNode* t;
        while(!Q.empty())
        {
            t=Q.front();
            Q.pop();
            t->left=nullptr;
            r->right=t;
            r=t;
        }
        root=dummyhead.right;
        return ;
    }
};
```
