# 字母异位词分组
[字母异位词分组](https://leetcode.cn/problems/group-anagrams/?envType=study-plan-v2&envId=top-100-liked)
作者确实不喜欢处理字符串的字母

```
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        vector <int> used(strs.size(),0); //统计数组
        vector <vector<string>> result; //结果

        for(int i=0;i<strs.size();i++){ //对于字符串遍历
            if(used[i]==1) //剪枝
                continue;
            used[i]=1;

            vector <string> sameword;//收集当前字母异位词
            sameword.push_back(strs[i]);

            vector <int> hash(26,0); //哈希表统计字母

            for(char c:strs[i]){
                hash[c-'a']++;
            }

            for(int j=i+1;j<strs.size();j++){
                if(strs[i].size()!=strs[j].size()) //剪枝
                    continue;

                vector <int> hashcopy=hash; //复制一份哈希表防止影响到下一次统计
                int wordsize=strs[i].size();
                for(char c:strs[j]){
                    if(hashcopy[c-'a']>0){ //str[i]中存在对应字母
                        hashcopy[c-'a']--;
                        wordsize--;
                    }
                    else //不存在直接下一个
                        continue;
                }
                if(wordsize==0){
                    sameword.push_back(strs[j]);
                    used[j]=1;
                    }
            }

            result.push_back(sameword);
        }

        return result;
    }
};
```
