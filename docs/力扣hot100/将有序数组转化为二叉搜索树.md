# 将有序数组转化为二叉搜索树
[将有序数组转换为二叉搜索树](https://leetcode.cn/problems/convert-sorted-array-to-binary-search-tree/description/?envType=study-plan-v2&envId=top-100-liked)

写起来更像是二分查找，如果各位在写过二分查找的搜索树就会发现二分查找的搜索树就是平衡二叉树

看起来我们在构建二叉树，不过更像是在反向二分查找？

各位如果写过二叉搜索树的构建，这道题思路也是类似的

```
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void PlantTree(TreeNode*&root,vector<int>& nums,int low,int high)
    {
        if(low<=high){
        int mid=(low+high)/2;
        root=new TreeNode(nums[mid]);//从中间开始创建
        PlantTree(root->left,nums,low,mid-1);//连接并创建左右子树
        PlantTree(root->right,nums,mid+1,high);
        }
    }
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        int high=nums.size()-1;
        TreeNode* root=nullptr;
        PlantTree(root,nums,0,high);
        return root;
    }
};
```
