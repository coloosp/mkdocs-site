# 合并两个有序链表

使用双指针和尾插法实现，依旧是链表基本题目

不过因为力扣给的都是不带头结点的链表，所以可以手搓一个

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        //众所周知力扣是不会给头结点的，所以各位手搓一个头结点比较好
        //使用尾插法来实现
        ListNode* H =new ListNode;

        ListNode* r=H;//尾节点

        ListNode* t;//暂存节点
        ListNode*p1=list1;
        ListNode*p2=list2;
        while(p1!=nullptr && p2!=nullptr){//将较小的插入尾部
            if(p1->val <p2->val){
                t=p1;
                p1=p1->next;
                r->next=t;
                r=t;
            }
            else{
                t=p2;
                p2=p2->next;
                r->next=t;
                r=t;
            }
        }

        while(p1!=nullptr){
             t=p1;
                p1=p1->next;
                r->next=t;
                r=t;
        }
        while(p2!=nullptr){//两个while只有一个会执行
              t=p2;
                p2=p2->next;
                r->next=t;
                r=t;
        }

            r->next=nullptr;

            ListNode *result=H->next;
            delete H;//养成好习惯
            return result;
    }
};
```
时间复杂度O(n)
空间复杂度O(1)
