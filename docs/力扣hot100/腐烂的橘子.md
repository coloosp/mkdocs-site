# 腐烂的橘子
[腐烂的橘子](https://leetcode.cn/problems/rotting-oranges/description/?envType=study-plan-v2&envId=top-100-liked)

看起来像什么生化危机一样

本题直观的思维应该就是广度优先搜索(不过听说有人也使用深搜回溯的方式完成了)

各位还记得我们的层序遍历吗
[二叉树的层序遍历](https://leetcode.cn/problems/binary-tree-level-order-traversal/?envType=study-plan-v2&envId=top-100-liked)

使用广搜/层序遍历时如何保证该轮需要弹出的节点数呢？或者完成了前面题目的各位应该知道怎么做了

值得一提的应该是，本题因为我们会把新鲜橘子1变成烂橘子2所有不需要标记数组，因为我们只会加入新一轮被腐烂的橘子



```
class Solution {
public:
    int result=0;
    int dir[4][2]={{1,0},{-1,0},{0,1},{0,-1}}; //方向数组

    void BFS(vector<vector<int>>&grid,int m,int n,int&fresh,queue<pair<int,int>>&Q)
    {
        bool flag =true;
        while(!Q.empty() && fresh >0)
        {
            result++;
            int level_size=Q.size(); //该轮腐烂橘子数
            while(level_size--){ //弹出并进行下一轮
            pair<int,int> node;
            node=Q.front();
            Q.pop();
            int curx=node.first;
            int cury=node.second;
            for(int i=0;i<4;i++)
            {
                int newx=curx+dir[i][0];
                int newy=cury+dir[i][1];

                //超界判断
                if(newx<0 || newx >=m || newy <0 || newy >=n)
                    continue;

                //只需要传染新鲜的！
                if(grid[newx][newy]==1)
                    {
                        fresh--;
                        grid[newx][newy]=2;//传染！
                        Q.push({newx,newy});
                    }
            }
            }
        }
    }

    int orangesRotting(vector<vector<int>>& grid) {
        //本题需要使用广搜来实现
        int fresh=0;
        int m=grid.size();
        int n=grid[0].size();
        queue <pair<int,int>> Q;//辅助队列
        
        //先统计新鲜橘子数
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j]==1)
                    fresh++;
            }
        }

        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                //将初始腐烂橘子都加入队列
                if(grid[i][j]==2)
                    {
                        Q.push({i,j});
                    }
            }
        }

        //本来就没有新鲜的
        if(fresh==0)
            return result;

        BFS(grid,m,n,fresh,Q);

        if(fresh >0)
            return -1;
        
        return result;

    }
};
```

时间复杂度O(mn)
空间复杂度O(mn)
